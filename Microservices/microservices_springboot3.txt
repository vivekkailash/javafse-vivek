@Entity

    @Id

    private String name;
    private String email;

    // getters + setters

    public String getName() { return name; }
    public void setName(String name) { this.name = name; }
    public String getEmail() { return email; }
    public void setEmail(String email) { this.email = email; }
}

UserController.java

package com.vivek.user.controller;

import java.util.List;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;
import com.vivek.user.model.User;
import com.vivek.user.repo.UserRepository;

@RestController
@RequestMapping("/users")
public class UserController {

    @Autowired
    private UserRepository userRepo;

    @GetMapping
    public List<User> all() { return userRepo.findAll(); }

    @GetMapping("/{id}")
    public User one(@PathVariable Long id) {
        return userRepo.findById(id).orElse(null);
    }

    @PostMapping
    public User create(@RequestBody User u) { return userRepo.save(u); }
}

OrderController.java
--------------------
package com.vivek.order.controller;

import java.util.List;
import java.util.Map;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;
import org.springframework.web.reactive.function.client.WebClient;

@RestController
@RequestMapping("/orders")
public class OrderController {

    @Autowired
    private OrderRepository orderRepo;

    @Autowired
    private WebClient userWebClient;

    @PostMapping
    public Order create(@RequestBody Order order) {
        // check if user exists in user-service before creating order
        Map user = userWebClient.get()
                .uri("/users/{id}", order.getUserId())
                .retrieve()
                .bodyToMono(Map.class)
                .block();

        if (user == null) {
            throw new RuntimeException("User not found");
        }
        return orderRepo.save(order);
    }

    @GetMapping
    public List<Order> all() { return orderRepo.findAll(); }
}

InventoryController.java

@RestController
@RequestMapping("/inventory")
public class InventoryController {

    @Autowired
    private RestTemplate restTemplate;

    @GetMapping("/{productId}")
    public Map checkStock(@PathVariable Long productId) {
        // use service name from eureka, not hostname
        String url = "http://PRODUCT-SERVICE/products/" + productId;
        Product p = restTemplate.getForObject(url, Product.class);

        Map result = new HashMap<>();
        result.put("productId", productId);
        result.put("stock", p.getStock());
        result.put("available", p.getStock() > 0);
        return result;
    }
}

public class RestTemplateConfig {

    @Bean
    @LoadBalanced          // <-- important, resolves service names via eureka
    public RestTemplate restTemplate() {
        return new RestTemplate();
    }

            - Path=/api/customers/**

            - RewritePath=/api/customers/(?<segment>.*), /customers/${segment}
            - name: RequestRateLimiter
              args:
                redis-rate-limiter.replenishRate: 5
                redis-rate-limiter.burstCapacity: 10
                key-resolver: "#{@userKeyResolver}"

        - id: billing_route
          uri: http://localhost:8102
          predicates:
            - Path=/api/billing/**
          filters:
            - RewritePath=/api/billing/(?<segment>.*), /billing/${segment}

  # for the rate limiter cache
  data:
    redis:
      host: localhost
      port: 6379


RateLimitConfig.java
--------------------
@Configuration
public class RateLimitConfig {

    @Bean
    public KeyResolver userKeyResolver() {
        // rate limit per user - here just use IP as demo
        return exchange -> Mono.just(
            exchange.getRequest().getRemoteAddress().getAddress().getHostAddress()
        );
    }
}

Test flow:
- request goes to  http://localhost:9000/api/customers/1
- gateway rewrites to  /customers/1
- forwards to  http://localhost:8101/customers/1
- if 5 requests in a second happen, rate limiter kicks in
- 11th request in the burst window -> 429 Too Many Requests

Exercise 4 - Circuit Breaker with Resilience4j
----------------------------------------------

** payment-service calls a slow third-party api **
** wrap with circuit breaker so we fail fast when it's down **


pom.xml (add)
-------------
<dependency>
    <groupId>org.springframework.cloud</groupId>
    <artifactId>spring-cloud-starter-circuitbreaker-resilience4j</artifactId>
</dependency>
<dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-aop</artifactId>
</dependency>


application.yml
---------------
resilience4j:
  circuitbreaker:
    instances:
      paymentApi:
        registerHealthIndicator: true
        slidingWindowSize: 10
        minimumNumberOfCalls: 5
        failureRateThreshold: 50
        waitDurationInOpenState: 10s
        permittedNumberOfCallsInHalfOpenState: 3

  timelimiter:
    instances:
      paymentApi:
        timeoutDuration: 2s


PaymentService.java
-------------------
package com.vivek.payment.service;

import io.github.resilience4j.circuitbreaker.annotation.CircuitBreaker;
import io.github.resilience4j.timelimiter.annotation.TimeLimiter;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.web.client.RestTemplate;

import java.util.concurrent.CompletableFuture;

@Service
public class PaymentService {

    Logger logger = LoggerFactory.getLogger(PaymentService.class);

    @Autowired
    private RestTemplate restTemplate;

    // wrap third party call with circuit breaker
    @CircuitBreaker(name = "paymentApi", fallbackMethod = "fallback")
    @TimeLimiter(name = "paymentApi")
    public CompletableFuture<String> processPayment(String orderId) {
        return CompletableFuture.supplyAsync(() -> {
            // call slow third party
            String url = "http://third-party-api.com/pay?order=" + orderId;
            return restTemplate.getForObject(url, String.class);
        });
    }

    // fallback when the api is down or times out
    public CompletableFuture<String> fallback(String orderId, Throwable ex) {
        logger.warn("Fallback triggered for order {} : {}", orderId, ex.getMessage());
        return CompletableFuture.completedFuture(
            "Payment queued for later processing. Order: " + orderId);
    }
}


PaymentController.java
----------------------
@RestController
public class PaymentController {

    @Autowired
    private PaymentService paymentService;

    @GetMapping("/pay/{orderId}")
    public CompletableFuture<String> pay(@PathVariable String orderId) {
        return paymentService.processPayment(orderId);
    }
}


Test:
GET  http://localhost:8080/pay/ORD100

** normal case - third party is up **
Output:
Payment successful for ORD100

** third party is slow / down **
Output (after ~2s timeout):
Payment queued for later processing. Order: ORD100

Log:
Fallback triggered for order ORD100 : TimeoutException


** circuit breaker states: **
** CLOSED  - normal, requests flow through **
** OPEN    - too many failures, fail fast without calling **
** HALF_OPEN - after wait time, try a few requests to see if it recovered **
